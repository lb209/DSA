  <script>
function node(value) {
 return{
     data:value,
    next:null
 }
}
let head=node(45)
head.next=node(35)
head.next.next=node(25)
head.next.next.next=node(5)

function print(head) {
    let current=head
    while(current !==null){
       console.log(current.data)
current=current.next
    }
   
}
function addNode(head,value) {
  let newNode=node(value)
  let current=head
  while(current.next !==null){
    current=current.next
  }
  current.next=newNode
}

function searc(head,value) {
  let current=head
  while (current !==null) {
    if(current.data===value){
      console.log(value,"ملا!")
      return
    }
current=current.next
  }
   console.log(value,"نہیں ملا!")
}
searc(head,33)
addNode(head, 100);
addNode(head, 111)
print(head);
